SDL -> sdl.node
  init -> SDL_Init
  initSubSystem -> SDL_InitSubSystem
  quit -> SDL_Quit
  quitSubSystem -> SDL_QuitSubSystem
  wasInit -> SDL_WasInit
  clearError -> SDL_ClearError
  getError -> SDL_GetError
  setError -> SDL_SetError
  waitEvent -> SDL_WaitEvent
  pollEvent -> SDL_PollEvent
  INIT
    TIMER -> SDL_INIT_TIMER timer subsystem
    AUDIO -> SDL_INIT_AUDIO audio subsystem
    VIDEO -> SDL_INIT_VIDEO video subsystem
    JOYSTICK -> SDL_INIT_JOYSTICK joystick subsystem
    EVERYTHING -> SDL_INIT_EVERYTHING all of the above subsystems
    NOPARACHUTE -> SDL_INIT_NOPARACHUTE don't catch fatal signals
